**ADVANCED INTERACTIONS:**
```javascript
// Cursor intimacy effect
document.addEventListener('mousemove', (e) => {
  const cursor = document.querySelector('.custom-cursor');
  const x = e.clientX;
  const y = e.clientY;
  
  // Smooth trailing effect
  gsap.to(cursor, {
    x: x - 10,
    y: y - 10,
    duration: 0.5,
    ease: 'expo.out'
  });

  // Interactive elements glow on proximity
  const elements = document.querySelectorAll('.interactive');
  elements.forEach(el => {
    const rect = el.getBoundingClientRect();
    const distance = Math.sqrt(
      Math.pow(x - (rect.left + rect.width/2), 2) + 
      Math.pow(y - (rect.top + rect.height/2), 2)
    );
    
    if (distance < 150) {
      el.style.boxShadow = `0 0 ${30 - distance/5}px rgba(216, 111, 56, ${0.5 - distance/300})`;
    }
  });
});
```

    initial="initial"
    animate="animate"
    exit="exit"
    variants={{
      initial: { opacity: 0, y: 50 },
      animate: { opacity: 1, y: 0 },
      exit: { opacity: 0, y: -20 }
